132 void ClientHandler::ReadSessionHeader1() {
133  boost::asio::async_read(m_TCPSocket, boost::asio::buffer(m_ReadBuffer, 3),[this](boost::system::error_code a_ErrorCode, std::size_t length) {
134  if (a_ErrorCode == boost::asio::error::operation_aborted) return;
135  if (!a_ErrorCode) {
136  // Check version of the access protocol
137  unsigned char l_Version = m_ReadBuffer[0];
138  if (l_Version != 0x00) {
139  // Unknown version of the access protocol
140  std::cerr << "Unsupported access protocol version rejected: " << (int)l_Version << std::endl;
141  Stop();
142  return;
143  } // if
144 
145  // Check service access point specifier: type of data
146  unsigned char l_SAP = m_ReadBuffer[1];
147  switch (l_SAP & 0xF0) {
148  case 0x00: {
149  m_eBufferType = BUFFER_TYPE_PAYLOAD;
150  break;
151  }
152  case 0x10: {
153  m_eBufferType = BUFFER_TYPE_PORT_STATUS;
154  break;
155  }
156  case 0x20: {
157  m_eBufferType = BUFFER_TYPE_PAYLOAD;
158  break;
159  }
160  case 0x30: {
161  m_eBufferType = BUFFER_TYPE_RAW;
162  break;
163  }
164  case 0x40: {
165  m_eBufferType = BUFFER_TYPE_DISSECTED;
166  break;
167  }
168  default:
169  // Unknown session type
170  std::cerr << "Unknown session type rejected: " << (int)(l_SAP & 0xF0) << std::endl;
171  Stop();
172  return;
173  } // switch
174 
175  // Check service access point specifier: reserved bit
176  if (l_SAP & 0x08) {
177  // The reserved bit was set... aborting
178  std::cerr << "Invalid reserved bit within SAP specifier of session header: " << (int)l_SAP << std::endl;
179  Stop();
180  return;
181  } // if
182 
183  // Check service access point specifier: invalids, deliver sent, and deliver rcvd
184  m_bDeliverInvalidData = (l_SAP & 0x04);
185  m_bDeliverSent = (l_SAP & 0x02);
186  m_bDeliverRcvd = (l_SAP & 0x01);
187 
188  // Now read the name of the serial port
189  ReadSessionHeader2(m_ReadBuffer[2]);
190  } else {
191  std::cerr << "TCP READ ERROR HEADER1:" << a_ErrorCode << std::endl;
192  Stop();
193  } // else
194  });

242 void ClientHandler::OnCtrlReceived(const PacketCtrl& a_PacketCtrl) {
243  // Check control packet: suspend / resume? Port kill request?
244  switch(a_PacketCtrl.GetPacketType()) {
245  case PacketCtrl::CTRL_TYPE_PORT_STATUS: {
246  if (a_PacketCtrl.GetDesiredLockState()) {
247  // Acquire a lock, if not already held. On acquisition, suspend the serial port
248  m_LockGuard.AcquireLock();
249  } else {
250  // Release lock and resume serial port
251  m_LockGuard.ReleaseLock();
252  } // else
253  break;
254  }
255  case PacketCtrl::CTRL_TYPE_ECHO: {
256  // Respond with an echo reply control packet: simply send it back
257  m_PacketEndpoint->Send(&a_PacketCtrl);
258  break;
259  }
260  case PacketCtrl::CTRL_TYPE_PORT_KILL: {
261  // Kill the serial port immediately and detach all related clients
262  m_SerialPortHandler->Stop();
263  break;
264  }
265  default:
266  break;
267  } // switch
268 }
