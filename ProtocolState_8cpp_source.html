136  // Checks
137  if (!m_bStarted) {
138  return;
139  } // if
140 
141  // Deliver raw frame to clients that have interest
142  if (m_SerialPortHandler->RequiresBufferType(BUFFER_TYPE_RAW)) {
143  m_SerialPortHandler->DeliverBufferToClients(BUFFER_TYPE_RAW, a_Payload, false, a_bMessageInvalid, false); // not escaped
144  } // if
145 
146  if (m_SerialPortHandler->RequiresBufferType(BUFFER_TYPE_DISSECTED)) {
147  m_SerialPortHandler->DeliverBufferToClients(BUFFER_TYPE_DISSECTED, a_Frame.Dissect(), false, a_bMessageInvalid, false);
148  } // if
149 
150  // Stop here if the frame was considered broken
151  if (a_bMessageInvalid) {
152  return;
153  } // if
154 
155  // A valid frame was received
156  if (m_AliveState->OnFrameReceived()) {
157  m_SerialPortHandler->PropagateSerialPortState();
158  } // if
159 
160  // Go ahead interpreting the frame we received
161  if (a_Frame.HasPayload()) {
162  // I-Frame or U-Frame with UI
163  if (m_SerialPortHandler->RequiresBufferType(BUFFER_TYPE_PAYLOAD)) {
164  m_SerialPortHandler->DeliverBufferToClients(BUFFER_TYPE_PAYLOAD, a_Frame.GetPayload(), a_Frame.IsIFrame(), a_bMessageInvalid, false);
165  } // if
166 
167  // If it is an I-Frame, the data may have to be acked
168  if (a_Frame.IsIFrame()) {
169  if (m_RSeqIncoming != a_Frame.GetSSeq()) {
170  m_SREJs.clear();
171  for (unsigned char it = m_RSeqIncoming; (it & 0x07) != a_Frame.GetSSeq(); ++it) {
172  m_SREJs.emplace_back(it);
173  } // for
174  } // if
175 
176  m_RSeqIncoming = ((a_Frame.GetSSeq() + 1) & 0x07);
177  m_bPeerRequiresAck = true;
178  } // if
179  } // if
180 
181  // Check the various types of ACKs and NACKs
182  if ((a_Frame.IsIFrame()) || (a_Frame.IsSFrame())) {
183  if ((a_Frame.IsIFrame()) || (a_Frame.GetHDLCFrameType() == Frame::HDLC_FRAMETYPE_S_RR)) {
184  if ((m_bPeerStoppedFlow) && (a_Frame.GetHDLCFrameType() == Frame::HDLC_FRAMETYPE_S_RR)) {
185  // The peer restarted the flow: RR clears RNR condition
186  m_bPeerStoppedFlow = false;
187  m_bWaitForAck = false;
188  m_Timer.cancel();
189  } // if
190 
191  // Currently, we send only one I-frame at a time and wait until the respective ACK is received
192  if ((m_bWaitForAck) && (a_Frame.GetRSeq() == ((m_SSeqOutgoing + 1) & 0x07))) {
193  // We found the respective sequence number to the last transmitted I-frame
194  m_bWaitForAck = false;
195  m_Timer.cancel();
196  m_WaitQueueReliable.pop_front();
197  } // if
198 
199  m_SSeqOutgoing = a_Frame.GetRSeq();
200  } else if (a_Frame.GetHDLCFrameType() == Frame::HDLC_FRAMETYPE_S_RNR) {
201  // The peer wants us to stop sending subsequent data
202  if (!m_bPeerStoppedFlow) {
203  // Start periodical query
204  m_bPeerStoppedFlow = true;
205  m_bPeerStoppedFlowNew = true;
206  m_bPeerStoppedFlowQueried = false;
207  } // if
208 
209  if (m_bWaitForAck) {
210  m_bWaitForAck = false;
211  m_Timer.cancel();
212  if (a_Frame.GetRSeq() == ((m_SSeqOutgoing + 1) & 0x07)) {
213  // We found the respective sequence number to the last transmitted I-frame
214  m_WaitQueueReliable.pop_front();
215  } // if
216  } // if
217 
218  // Now we know which SeqNr the peer awaits next... after the RNR condition was cleared
219  m_SSeqOutgoing = a_Frame.GetRSeq();
220  } else if (a_Frame.GetHDLCFrameType() == Frame::HDLC_FRAMETYPE_S_REJ) {
221  if (m_bPeerStoppedFlow) {
222  // The peer restarted the flow: REJ clears RNR condition
223  m_bPeerStoppedFlow = false;
224  m_bWaitForAck = false;
225  m_Timer.cancel();
226  } // if
227 
228  // The peer requests for go-back-N. We have to retransmit all affected packets, but not with this version of HDLC.
229  if (a_Frame.GetRSeq() == m_SSeqOutgoing) {
230  // We found the respective sequence number to the last transmitted I-frame
231  m_bWaitForAck = false;
232  m_Timer.cancel();
233  } // if
234 
235  m_SSeqOutgoing = ((a_Frame.GetRSeq() + 0x07) & 0x07);
236  } else {
237  assert(a_Frame.GetHDLCFrameType() == Frame::HDLC_FRAMETYPE_S_SREJ);
238  if (m_bPeerStoppedFlow) {
239  // The peer restarted the flow: SREJ clears RNR condition
240  m_bPeerStoppedFlow = false;
241  m_bWaitForAck = false;
242  m_Timer.cancel();
243  } // if
244 
245  // The peer requests for the retransmission of a single segment with a specific sequence number
246  // This cannot be implemented using this reduced version of HDLC!
247  if ((m_bWaitForAck) && (a_Frame.GetRSeq() == m_SSeqOutgoing)) {
248  // We found the respective sequence number to the last transmitted I-frame.
249  // In this version of HDLC, this should not happen!
250  m_bWaitForAck = false;
251  m_Timer.cancel();
252  } // if
253  } // else
254  } // if
255

257  // Check if we have to send something now
258  OpportunityForTransmission();
259  } // if

262 void ProtocolState::OpportunityForTransmission() {
263  // Checks
264  if (!m_bAwaitsNextHDLCFrame) {
265  return;
266  } // if
267 
268  Frame l_Frame;
269  if (m_bSendProbe) {
270  // The correct baud rate setting is unknown yet, or it has to be checked again. Send an U-TEST frame.
271  m_bSendProbe = false;
272  l_Frame = PrepareUFrameTEST();
273  } // if
274 
275  if (l_Frame.IsEmpty() && m_AliveState->IsAlive()) {
276  // The serial link to the device is alive. Send all outstanding S-SREJs first.
277  if (m_SREJs.empty() == false) {
278  // Send SREJs first
279  l_Frame = PrepareSFrameSREJ();
280  } // if
281 
282  // Check if packets are waiting for reliable transmission
283  if (l_Frame.IsEmpty() && (m_WaitQueueReliable.empty() == false) && (!m_bWaitForAck) && (!m_bPeerStoppedFlow)) {
284  // Send an I-Frame now
285  l_Frame = PrepareIFrame();
286  m_bWaitForAck = true;
287 
288  // I-frames carry an ACK
289  m_bPeerRequiresAck = false;
290 
291  // Start retransmission timer
292  auto self(shared_from_this());
293  m_Timer.expires_from_now(boost::posix_time::milliseconds(500));
294  m_Timer.async_wait([this, self](const boost::system::error_code& ec) {
295  if (!ec) {
296  // Send the head element of the wait queue again
297  m_bWaitForAck = false;
298  OpportunityForTransmission();
299  } // if
300  });
301  } // if
302 
303  // Send outstanding RR?
304  if (l_Frame.IsEmpty() && ((m_bPeerRequiresAck) || m_bPeerStoppedFlowNew)) {
305  bool l_bStartTimer = false;
306  if (m_bPeerStoppedFlowNew) {
307  m_bPeerStoppedFlowNew = false;
308  l_bStartTimer = true;
309  } else {
310  // Prepare RR
311  l_Frame = PrepareSFrameRR();
312  m_bPeerRequiresAck = false;
313  if (m_bPeerStoppedFlow && !m_bPeerStoppedFlowQueried) {
314  // During the RNR confition at the peer, we query it periodically
315  l_Frame.SetPF(true); // This is a hack due to missing command/response support
316  m_bPeerStoppedFlowQueried = true;
317  l_bStartTimer = true;
318  } // if
319  } // else
320 
321  if (l_bStartTimer) {
322  m_Timer.cancel();
323  auto self(shared_from_this());
324  m_Timer.expires_from_now(boost::posix_time::milliseconds(500));
325  m_Timer.async_wait([this, self](const boost::system::error_code& ec) {
326  if (!ec) {
327  if (m_bPeerStoppedFlow) {
328  m_bPeerRequiresAck = true;
329  m_bPeerStoppedFlowQueried = false;
330  OpportunityForTransmission();
331  } // if
332  } // if
333  });
334  } // if
335  } // if
336 
337  // Check if packets are waiting for unreliable transmission
338  if (l_Frame.IsEmpty() && (m_WaitQueueUnreliable.empty() == false)) {
339  l_Frame = PrepareUFrameUI();
340  m_WaitQueueUnreliable.pop_front();
341  } // if
342 
343  // If there is nothing to send, try to fill the wait queues, but only if necessary.
344  if (l_Frame.IsEmpty()) {
345  // These expressions are the result of some boolean logic
346  bool l_bQueryReliable = (m_WaitQueueUnreliable.empty() && m_WaitQueueReliable.empty() && (!m_bPeerStoppedFlow));
347  bool l_bQueryUnreliable = (m_WaitQueueUnreliable.empty() && (m_WaitQueueReliable.empty() || m_bPeerStoppedFlow));
348  if (l_bQueryReliable || l_bQueryUnreliable) {
349  m_SerialPortHandler->QueryForPayload(l_bQueryReliable, l_bQueryUnreliable);
350  } // if
351  } // if
352  } // if
353 
354  if (l_Frame.IsEmpty() == false) {
355  // Deliver unescaped frame to clients that have interest
356  m_bAwaitsNextHDLCFrame = false;
357  auto l_HDLCFrameBuffer = FrameGenerator::SerializeFrame(l_Frame);
358  if (m_SerialPortHandler->RequiresBufferType(BUFFER_TYPE_RAW)) {
359  m_SerialPortHandler->DeliverBufferToClients(BUFFER_TYPE_RAW, l_HDLCFrameBuffer, l_Frame.IsIFrame(), false, true); // not escaped
360  } // if
361 
362  if (m_SerialPortHandler->RequiresBufferType(BUFFER_TYPE_DISSECTED)) {
363  m_SerialPortHandler->DeliverBufferToClients(BUFFER_TYPE_DISSECTED, l_Frame.Dissect(), l_Frame.IsIFrame(), false, true);
364  } // if
365 
366  m_SerialPortHandler->TransmitHDLCFrame(std::move(FrameGenerator::EscapeFrame(l_HDLCFrameBuffer)));
367  } // if
